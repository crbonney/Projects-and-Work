70void* keyboardserver_thread(void* i)



71{



72    struct sched_param param;



73    param.sched_priority = 9;



74    pthread_setschedparam(pthread_self(), 2, &param);



75    printf("keyboardserver_thread(): Entered the Keyboard Server thread... \n");



76    



77    if((sockfd = socket(AF_INET, SOCK_STREAM, 0)) < 0) {



78        printf("keyboardserver_thread(): ERROR : Could not create socket \n");



79        exit(1);



80    }



81    



82    printf("keyboardserver_thread(): Successfully created socket... \n");



83    



84    memset(&serv_addr, '0', sizeof(serv_addr));



85    



86    serv_addr.sin_family = AF_INET;



87    serv_addr.sin_addr.s_addr = htonl(INADDR_ANY);



88    serv_addr.sin_port = htons(12391);



89    



90    printf("keyboardserver_thread(): Bind socket... \n");



91    



92    // Fix : We could not run the Spider Flight Continuously. We need to wait for some time before you start the next flight.



93    // To fix the issue we are creating Socket with REUSEADDR.



94    if (setsockopt(sockfd, SOL_SOCKET, SO_REUSEADDR, &yes, sizeof(int)) < 0)



95    {



96        printf("keyboardserver_thread(): ERROR: Failure in Reusing address.. with error: %d\n",errno);



97        exit(1);



98    }



99    



100    



101    if(bind(sockfd, (struct sockaddr *)&serv_addr, sizeof(serv_addr)) < 0) {



102        printf("keyboardserver_thread(): ERROR Connection to reference value server failed %d \n", errno);



103        exit(1);



104    }



105    else



106    {



107        printf("keyboardserver_thread(): Bind completed .. listening to the socket !\n");



108        



109        if (listen(sockfd, 10)<0)



110        {



111            printf("keyboardserver_thread(): Socket error (listen) %d\n",errno);



112            exit(1);



113        }



114        connfd = accept(sockfd, (struct sockaddr*)NULL, NULL);



115        setsockopt(connfd, SOL_SOCKET, SO_REUSEADDR, &yes, sizeof(yes));



116        setsockopt(connfd, SOL_SOCKET, SO_SNDBUF, &buff_size, sizeof(buff_size));



117        clientConnect = 1;



118        printf("keyboardserver_thread(): Received connection from the host! \n");



119    }



120    



121   ///////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////// 



122    



123     if((sockfd_GUI = socket(AF_INET, SOCK_STREAM, 0)) < 0) {



124        printf("keyboardserver_thread(): ERROR : Could not create GUI socket\n");



125        exit(1);



126    }



127    



128    printf("keyboardserver_thread(): Successfully created GUI socket... \n");



129    



130    memset(&serv_addr, '0', sizeof(serv_addr));



131    



132    serv_addr.sin_family = AF_INET;



133    serv_addr.sin_addr.s_addr = htonl(INADDR_ANY);



134    serv_addr.sin_port = htons(24099);



135    



136    printf("keyboardserver_thread(): Bind GUI socket...\n");



137    



138    // Fix : We could not run the Spider Flight Continuously. We need to wait for some time before you start the next flight.



139    // To fix the issue we are creating Socket with REUSEADDR.



140    if (setsockopt(sockfd_GUI, SOL_SOCKET, SO_REUSEADDR, &yes, sizeof(int)) < 0)



141    {



142        printf("keyboardserver_thread(): ERROR: Failure in Reusing address in GUI.. with error: %d\n",errno);



143        exit(1);



144    }



145    



146    



147    if(bind(sockfd_GUI, (struct sockaddr *)&serv_addr, sizeof(serv_addr)) < 0) {



148        printf("keyboardserver_thread(): ERROR Connection to GUI server failed %d \n", errno);



149        exit(1);



150    }



151    else



152    {



153        printf("keyboardserver_thread(): Bind completed .. listening to the GUI socket !\n");



154        



155        if (listen(sockfd_GUI, 10)<0)



156        {



157            printf("keyboardserver_thread(): GUI Socket error (listen) %d\n",errno);



158            exit(1);



159        }



160        connfd_GUI = accept(sockfd_GUI, (struct sockaddr*)NULL, NULL);



161        setsockopt(connfd_GUI, SOL_SOCKET, SO_REUSEADDR, &yes, sizeof(yes));



162        setsockopt(connfd_GUI, SOL_SOCKET, SO_SNDBUF, &buff_size, sizeof(buff_size));



163        printf("keyboardserver_thread(): Received connection to GUI from the host! \n");



164    }



165    



166   ////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////



167    return NULL;



168}

170void create_referencevalserver_thread(void)



171{



172    int err;



173    pthread_t img_tid;



174    int policy;



175    struct sched_param param;



176    pthread_attr_t attr;



177    size_t stacksize;



178    



179    pthread_attr_init(&attr);



180    pthread_attr_getstacksize(&attr,&stacksize);



181    err = pthread_attr_setstacksize(&attr,PTHREAD_STACK_MIN);



182    printf("create_referencevalserver_thread(): stack size before reference value thread creation %d err=%d\n", stacksize,err);



183    



184    img_tid = pthread_self();



185    printf("create_referencevalserver_thread(): reference value server thread id = %d \n",(int) img_tid);



186    pthread_getschedparam(img_tid, &policy,&param);



187    printf("create_referencevalserver_thread(): reference value server  policy = %d pri=%d\n", (int)policy,param.sched_priority);



188    



189    // param1.sched_priority = param.sched_priority - 1;



190    // pthread_setschedparam(tid,policy,&param1);



191    err = pthread_create(&tid, &attr, &keyboardserver_thread, NULL);



192    if (err != 0)



193        printf("create_referencevalserver_thread(): ERROR: can't create thread :[%s]", strerror(err));



194    else



195        printf("create_referencevalserver_thread(): Reference Server Thread created successfully\n");



196    



197    pthread_attr_getstacksize(&attr,&stacksize);



198    printf("create_referencevalserver_thread(): stacksize after reference thread creation %d \n", stacksize);



199}

208void* extmode_thread(void* i)



209{



210    static int counter = 0;



211    counter++;



212    



213    if (counter == 1)



214    {



215        int policy;



216        struct sched_param param ;



217        



218        pthread_getschedparam(pthread_self(), &policy,&param);



219        printf("extmode_thread(): External mode thread policy = %d and priority=%d\n", (int)policy,param.sched_priority);



220        



221        /* initialize external mode */



222        rtParseArgsForExtMode(0, NULL);



223        rtmSetErrorStatus(MW_StringifyDefineFunction(MODEL,_M), 0);



224        MW_StringifyDefineFunction(MODEL,_initialize)();



225        /* External mode */



226        rtSetTFinalForExtMode(&rtmGetTFinal(MW_StringifyDefineFunction(MODEL,_M)));



227        rtExtModeCheckInit(1);



228        rtERTExtModeStartMsg();



229    }



230    



231    while(runModel)



232    {



233        boolean_T rtmStopReq = false;



234        #ifdef MW_HAS_IMAGE_PROCESSING



235            rtExtModeOneStep(MW_StringifyDefineFunction(MODEL,_M->extModeInfo), 2, &rtmStopReq); // This is used when the Image processing is enabled along with multitasking. This will upload the step0() and step1() data.



236        #else



237            rtExtModeOneStep(MW_StringifyDefineFunction(MODEL,_M->extModeInfo), 1, &rtmStopReq); // This is used when only the control code is running. Multitasking is not enabled in this case and only the step() data is required to be uploaded.



238        #endif 



239        if (rtmStopReq)



240        {



241            printf("extmode_thread(): Thread received STOP command \n");



242            rtmSetStopRequested(MW_StringifyDefineFunction(MODEL,_M), true);



243            // Reason for the following code:



244            // If the user presses stop button before starting the flight code i.e before starting



245            // the execution step code.



246            rtExtModeShutdown(1);



247            MW_StringifyDefineFunction(MODEL,_terminate)();



248            runModel = 0;



249        }



250        



251        if (run_flag == 0)



252        {



253            rtExtModeShutdown(1);



254            MW_StringifyDefineFunction(MODEL,_terminate)();



255            runModel = 0;



256        }



257    }



258    



259    if (runModel == 0)



260    {



261        rtExtModeShutdown(1);



262        MW_StringifyDefineFunction(MODEL,_terminate)();



263    }



264    



265    background_task_completed = true;



266    return NULL;



267}

269void create_background_thread(void)



270{



271    int err, status;



272    pthread_attr_t attr;



273    size_t stacksize;



274    struct sched_param param;



275    



276    pthread_attr_init(&attr);



277    err = pthread_attr_setinheritsched(&attr,PTHREAD_EXPLICIT_SCHED);



278    printf("create_background_thread(): ext thread inherit scheduler option error=%d\n", err);



279    



280    status = pthread_attr_setdetachstate(&attr, PTHREAD_CREATE_DETACHED);



281    printf("create_background_thread(): value of detach status=%d\n", status);



282    



283    err = pthread_attr_setstacksize(&attr,PTHREAD_STACK_MIN);



284    printf("create_background_thread(): stacksize before thread creation err=%d\n", err);



285    



286    /* In case of Mambo , the policy of rsedu_control.c is set as 2 (SCHED_RR) and priority 32



287     * In case of Rolling Spider , the policy of rsedu_control.c is set as 0 (SCHED_OTHER) and priority 0



288     * Setting the policy and priority of External Mode thread as 0 (SCHED_OTHER) and priority 0 */



289    



290    status = pthread_attr_setschedpolicy(&attr, SCHED_OTHER);



291    printf("create_background_thread(): value of scheduler policy status=%d\n", status);



292



293    param.sched_priority = 0;



294    status = pthread_attr_setschedparam(&attr, &param);



295    printf("create_background_thread(): value of scheduler param status=%d\n", status);



296    



297    err = pthread_create(&ext_tid, &attr, &extmode_thread, NULL);



298    if (err != 0)



299        printf("create_background_thread(): can't extmode thread :[%s]", strerror(err));



300    else



301        printf("create_background_thread(): External mode Thread created successfully\n");



302    



303    pthread_attr_getstacksize(&attr,&stacksize);



304    printf("create_background_thread(): stacksize after ext thread creation %d \n", stacksize);



305    



306}
